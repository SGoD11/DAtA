#include
        #include
        #define MAX 20
        int top=-1,A[MAX];
        void push(){
            
            if(top==MAX-1){
                printf("Elements are full");
            }
            else{
                
                printf("Enter elements in the stack: ");
                top++;
                scanf("%d",&A[top]);
            }
            
            
        }
        void pop(){
            
            if(top==-1){
                printf("stack empty");
            }
            else{
                
                printf("Deleted element is : ");
                top--;
            }
            
        }
        void display(){
            int d = top;
            if(d<0){
                printf("No elements found ");
            }
            while(d>0){
                printf("%d\n",A[d]);
                d--;
            }
            
        }
        
        int main(){
            int ch;
            
            do{
            printf("Stack Program\n1.push\n2.pop\n3.display\n4.exit\nenter your choice: ");
            scanf("%d",&ch);
            switch(ch)
            {
                case 1 : push();break;
                case 2 : pop();break;
                case 3 : display();break;
                case 4 : return 0;
                default : printf("enter option correctly\n");	
            }
            }while(ch!=4);
            return 0;
            
            
        }
