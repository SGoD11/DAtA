            #define MAX 20
                        
            int front=-1,rear=-1,A[MAX];
                        
                        void insert(){
                            if(rear==MAX-1){
                                printf("Queue is full");
                            }
                            else
                            {
                                if(front==-1){
                                    front++;
                                }
                                rear++;
                                printf("enter element : " );
                                scanf("%d",&A[rear]);
                            
                            }
                            
                        }
                        
                        void del(){
                            if(front==-1){
                                printf("Queue is empty");
                                
                            }
                            
                                printf("deleted element is : %d",A[front]);
                                
                                if(front==rear){
                                    front=-1;
                                    rear=-1;
                                }
                                else
                                front++;
                            
                            
                            
                        }
                        
                        void display(){
                            int t =front;
                            if(t==-1){
                                printf("Queue is empty");
                            }
                            while(t<=rear){
                                printf("%d\n",A[t]);
                                t++;
                            }
                            
                        }
                        
                        int main(){
                            
                            int ch;
                            do{
                                printf("Queue Program\n1.Insert\n2.Delete\n3.display\n4.exit\nenter your choice: ");
                            scanf("%d",&ch);
                            switch(ch)
                            {
                                case 1 : insert();break;
                                case 2 : del();break;
                                case 3 : display();break;
                                case 4 : return 0;
                                default : printf("enter option correctly\n");	
                            }
                            }while(ch!=4);
